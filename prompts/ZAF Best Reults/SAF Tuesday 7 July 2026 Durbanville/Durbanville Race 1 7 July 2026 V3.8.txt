import numpy as np

def calculate_optimized_frailty(horse):
    """
    Ingests horse variables and applies GRSV, KEMI, and VOFF modifications
    to optimize latent frailty and Nominal Latent Speed (NLS).
    """
    # 1. Base Parameters
    base_frailty = 0.50
    rqv_mod = -(horse['rqv'] / (horse['rqv'] + 4.0))
    
    # 2. Programmatic Veterinary Obstruction Forgiveness Factor (VOFF)
    nls_base = horse['raw_nls']
    if horse['vet_obstruction_last_start'] == True and horse['obstruction_resolved'] == True:
        voff_bonus = 4.00  # Reinstate lost latent speed points
        nls_calibrated = nls_base + voff_bonus
    else:
        nls_calibrated = nls_base

    # 3. Geometric Rail Sparing Vector (GRSV)
    # Applied strictly on turning tracks (R_turn < 100m)
    if horse['track_turn_radius'] < 100.0:
        alpha_rail = 0.06
        beta_draw = 0.50
        delta_nu_grsv = -alpha_rail * np.exp(-beta_draw * (horse['barrier_actual'] - 1))
    else:
        delta_nu_grsv = 0.0

    # 4. Kinetic Experience Maturation Index (KEMI)
    # Applied to sprints (D <= 1200m) on Pathway A (Maidens/Juveniles)
    if horse['race_distance'] <= 1200 and horse['pathway'] == 'A':
        kappa_exp = 0.02
        kemi_scalar = 1.0 + kappa_exp * np.log(horse['career_starts'] + 1)
        # Apply the scalar to increase the raw speed capability
        nls_calibrated = nls_calibrated * kemi_scalar
    else:
        kemi_scalar = 1.0

    # 5. Compile Modified Frailty Variable
    dynamic_modifiers = (
        horse['delta_baseline'] +
        horse['delta_nu_decay'] +
        horse['delta_nu_mass'] +
        horse['delta_nu_geometry'] +  # Standard turning drag
        delta_nu_grsv +               # New GRSV Sparing Bonus
        horse['delta_nu_pedigree'] +
        horse['delta_nu_gear'] +
        horse['delta_nu_tempo'] +
        horse['delta_nu_stress'] +
        horse['delta_nu_age_sex'] +
        horse['delta_nu_spacing'] +
        horse['delta_nu_psb']
    )
    
    optimized_frailty = base_frailty + rqv_mod + dynamic_modifiers
    optimized_frailty = max(0.15, min(0.85, optimized_frailty)) # Safeguard bounds
    
    # Recalculate NTCI with the updated parameters
    optimized_ntci = nls_calibrated * (1.0 - optimized_frailty)
    
    return {
        'optimized_nls': nls_calibrated,
        'optimized_frailty': optimized_frailty,
        'optimized_ntci': optimized_ntci
    }
