SRD_{\text{stable}} = \frac{N_{\text{stable\_runners}}}{N_{\text{active\_field}}}

SIS_i = TDP_{\text{stable}} \times JBT_i \times \left(1.0 + \lambda \cdot SRD_{\text{stable}}\right)
